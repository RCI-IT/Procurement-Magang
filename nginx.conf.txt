events {}

http {
  server {
    listen 80;
    server_name procurement.rci;

    # üîÅ Redirect /hr ke /hr/
    location = /hr {
      return 301 /hr/;
    }

    # üì¶ HR Frontend
    # Procurement frontend (default)
    location / {
      proxy_pass http://procurementfrontend:3000;
      proxy_http_version 1.1;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # HR/Employee frontend
    location /hr/ {
      rewrite ^/hr(/.*)$ $1 break;  # remove /hr prefix for frontend routing
      proxy_pass http://employeefrontend:3000;
      proxy_http_version 1.1;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # üî∂ Procurement Backend API (prefix: /api/)
    location /api {
      rewrite ^/api/(.*)$ /$1 break;
      proxy_pass http://procurementbackend:5000/;
      proxy_http_version 1.1;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header X-Forwarded-Proto $scheme;

      # Optional: CORS headers
      add_header Access-Control-Allow-Origin *;
      add_header Access-Control-Allow-Methods 'GET, POST, OPTIONS, PUT, DELETE';
      add_header Access-Control-Allow-Headers 'Origin, Content-Type, Accept, Authorization';
    }

    # üî∑ HR Backend API (prefix: /apiHR/)
    location /apiHR {
      rewrite ^/apiHR/(.*)$ /$1 break;
      proxy_pass http://employeebackend:4000/;
      proxy_http_version 1.1;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header X-Forwarded-Proto $scheme;

      # Optional: CORS headers
      add_header Access-Control-Allow-Origin *;
      add_header Access-Control-Allow-Methods 'GET, POST, OPTIONS, PUT, DELETE';
      add_header Access-Control-Allow-Headers 'Origin, Content-Type, Accept, Authorization';
    }

    # Optional: fallback error page for SPA routing
    error_page 404 /index.html;
  }
}
